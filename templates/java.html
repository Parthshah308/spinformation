public class recursion {
    static int factorial(int n) {
       if(n==0 || n==1)
       {
           return 1;
       }

        else {
            return n * factorial(n - 1);
        }
    }

class Library{

         books=new String[100];
         numofbooks=0;
    }
    void addbook(String book)
    {
        books[numofbooks]=book;
        numofbooks++;
        System.out.println(book +" named book is added!!!");
    }
    void issuebook(String book)
    {int i;
        for(i=0;i<books.length;i++)
        {

                if(books[i]==book)
                {
                    System.out.println("this books has been issued");
                    books[i]=null;
                    return;
                }
            }
        System.out.println("This books is not available here...");
    }
    void returnbook(String book){
        addbook(book);
    }
    void showbooks(){
        System.out.println("The Available books are below here...");
        for (String book:books) {

            {
                continue;
            }
                System.out.println("* " + book);

        }
    }
}
