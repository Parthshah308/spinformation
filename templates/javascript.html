<!DOCTYPE html>
<html lang="en">
<head>
  <meta charset="UTF-8">
  <meta name="viewport" content="width=device-width, initial-scale=1.0">
  <meta http-equiv="X-UA-Compatible" content="ie=edge">
  <title>Cel to fah conveter</title>
</head>
<body>
    <form onsubmit="resultconverter(); return false">
      <p>Convert celsius to fahernhit</p><br>
      <input type="number" id="inputnum">
    <select id="inputfield" name="">
      <option value="cel">
        celsius
      </option>
      <option value="fah">
        fahernhit
      </option>
    </select><br><br><br>
<button type="submit">Submit</button>
    <p id="resultcon"></p>
  </form>
    <script type="text/javascript">
      const resultconverter = () =>{
      const inputnum=document.querySelector('#inputnum').value;
      // console.log(inputnum);

      const inputfild=document.querySelector('#inputfield');
      // console.log(inputfild);
      const valuetemp=inputfield.options[inputfild.selectedIndex].value;
      // console.log();
      const celtofah= (cel)=>{
        let celsius=Math.round((cel *9/5)+32);
        return celsius;
      }

      const fahtocel= (fah)=>{
        let fahernhit=Math.round((fah - 32) * 5/9);
        return fahernhit;
      }

      let result;
      if(valuetemp == 'cel')
      {
         result=celtofah(inputnum);
         document.getElementById('resultcon').innerHTML= `= ${result} fahernhit`;
      }
      else{
        result=fahtocel(inputnum);
        document.getElementById('resultcon').innerHTML= `= ${result} celsius`;
      }

    }
    </script>
</body>
</html>
