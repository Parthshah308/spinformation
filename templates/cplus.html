Program 44:Multilevel inheritance. 
#include<iostream>
using namespace std;
class student
{
  protected:
    int rollno;
  public:
    void setrollno(int a)
    {
      rollno=a;
    }
    void getrollno()
    {
      cout<<"Student rollno is="<<rollno<<endl;
    }
};

class marks:public student
{
  protected:
    float maths;
    float sciense;
  public:
    void setmarks(float c,float d)
    {
      maths=c;
      sciense=d;
    }
    void getmarks()
    {
      cout<<"This student maths marks is ="<<maths<<endl;
      cout<<"This student sciense marks is ="<<sciense<<endl;
      
    }
};

class result:public marks
{
  public:
    void show()
    {
      getrollno();
      getmarks();
      cout<<"This student result is ="<<(maths+sciense)/2<<endl;
    }
};
int main()
{
result p;
p.setrollno(1153);
p.setmarks(95.3,91.12);
p.show();
return 0;
}
